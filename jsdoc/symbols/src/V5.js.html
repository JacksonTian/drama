 28     V5.ready = function (callback) {
 29         if (document.readyState === "complete") {
 30             callback();
 31         } else {
 32             var ready = function () {
 33                 if (document.readyState === "complete") {
 34                     callback();
 35                     // Remove the callback listener from readystatechange event.
 36                     $(document).unbind("readystatechange", ready);
 37                 }
 38             };
 39             //  Bind callback to readystatechange
 40             $(document).bind("readystatechange", ready);
 41         }

 63             window.addEventListener("touchmove", function (e) {e.preventDefault(); }, false);
 64             // Use popstate to handle history go/back.
 65             window.addEventListener('popstate', function (event) {
 66                 var params = event.state;
 67                 if (params) {
 68                     var args = params.split("/");
 69                     var currentHash = args.shift();
 70                     console.log("Hash Changed: " + currentHash);
 71                     if (V5.hashHistory.length) {
 72                         console.log(V5.hashHistory);
 73                         if (currentHash !== undefined) {
 74                             var topHash = _.map(V5.hashMap, function (val, key) {
 75                                 return _.last(val);
 76                             });
 77                             if (_.include(topHash, params)) {
 78                                 console.log("changed, but no action.");
 79                             } else {
 80                                 var hashStack = _.compact(_.map(V5.hashMap, function (val, key) {
 81                                     return _.include(val, currentHash) ? key : null;
 82                                 }));
 83                                 console.log(hashStack);
 84                                 V5.hashMap[hashStack[0]].pop();
 85                                 V5.trigger("openView", currentHash, _.indexOf(V5.columns, hashStack[0]));
 86                                 console.log("Forward or back");
 87                             }
 88                         }
 89                     }
 90                 }
 91             }, false);
 92 
 93             // Handle refresh case or first visit.
 94             if (V5.hashHistory.length === 0) {
 95                 var map = V5.hashMap;
 96                 if (_.size(map)) {
 97                     // Restore view from session.
 98                     console.log("Restore from session.");
 99                     V5.restoreViews();
100                 } else {
101                     // Init view.
102                     console.log("Init view.");
103                     V5.initView();
104                 }
105             }
106         });

168     /**
169      * @description Gets View from cache or server. If the view file come from server, 
170      * the callback will be executed async, and cache it.
171      * @param {string} viewName View name.
172      * @param {boolean} enableL10N Flag whether this view's localization enabled.
173      * If true, will generate view with localization resources.
174      * @param {function} callback Callback function, will be called after got the view from cache or server.
175      */
176     V5.getView = function (viewName, enableL10N, callback) {
177         var _pageCache = V5._pageCache;
178         var page = V5._pages[viewName];
179         var proxy = new EventProxy();
180         proxy.assign("l10n", "view", function (l10n, view) {
181             var html = V5.localize(view, l10n);
182             page.resources = l10n;
183             callback($(html));
184         });
185 
186         if (_pageCache[viewName]) {
187             proxy.trigger("view", _pageCache[viewName]);
188         } else {
189             $.get("pages/" + viewName + ".page?_=" + new Date().getTime(), function (text) {
190                 // Save into cache.
191                 _pageCache[viewName] = text;
192                 proxy.trigger("view", _pageCache[viewName]);
193             });
194         }
195 
196         // Fetch the localize resources.
197         if (page.enableL10N) {
198             V5.fetchL10N(viewName, function () {
199                 proxy.trigger("l10n", V5.L10N[V5.langCode][viewName]);
200             });
201         } else {
202             proxy.trigger("l10n", {});
203         }
204     };

206     /**
207      * @description Display view in view column.
208      * @private
209      * @param {string} hash View hash, view name.
210      * @param {number} effectColumn View column's index.
211      * @param {array} args Parameters of view.
212      * @param {object} viewport Which viewport, if don't set, will use default viewport.
213      */
214     V5.displayView = function (hash, effectColumn, args, viewport) {
215         var columnName = V5.columns[effectColumn];
216         var column = viewport.find("." + columnName);
217         if (column.size() < 1) {
218             column = $("<div><div class='column_loading'><div class='loading_animation'></div></div></div>");
219             column.addClass(columnName);
220             viewport.append(column);
221         }
222 
223         if (viewport === V5.viewport) {
224             if (V5.hashMap[columnName]) {
225                 V5.hashMap[columnName].push([hash].concat(args).join("/"));
226             } else {
227                 V5.hashMap[columnName] = [[hash].concat(args).join("/")];
228             }
229             V5.hashHistory.push([hash].concat(args));
230         }
231 
232         var page = V5._pages[hash];
233         if (page) {
234             var previousPage = column.find("section.page.active").removeClass("active");
235             if (previousPage.length) {
236                 var id = previousPage.attr('id');
237                 var previous = V5._pages[id];
238                 if (previous && id !== hash) {
239                     previous.shrink();
240                 }
241             }
242 
243             var loadingNode = column.find(".column_loading").removeClass("hidden");
244             var page = V5._pages[hash];
245             V5.getView(hash, page.enableL10N, function (view) {
246                 loadingNode.addClass("hidden");
247                 if (viewport === V5.viewport) {
248                     viewport.attr("class", V5.columnModes[_.size(V5.hashMap) - 1]);
249                 }
250                 page.columnIndex = _.indexOf(V5.columns, columnName);
251                 if (!page.initialized) {
252                     column.append(view);
253                     page.node = view;
254                     page.node.addClass("active");
255                     page.initialize.apply(page, args);
256                     page.initialized = true;
257                 } else {
258                     if (page.parameters.toString() !== args.toString()) {
259                         page.destroy();
260                         page.node.remove();
261                         page.initialized = false;
262                         V5.getView(hash, page.enableL10N, arguments.callee);
263                         return;
264                     } else {
265                         page.node.addClass("active");
266                         page.reappear();
267                     }
268                 }
269 
270                 page.parameters = args;
271                 page.viewport = viewport;
272             });
273         } else {
274             throw hash + " module doesn't be defined.";
275         }
276     };

286     V5.hashMap = (function () {
287         var session = getStorage("session");
288         var hashMap = session.get("hashMap");
289         if (!hashMap) {
290             hashMap = {};
291         } else {

295             session.put("hashMap", V5.hashMap);
296         });

308      * @description A factory method to generate View object. Packaged on Backbone.View.
309      * @param {node} node a $(Zepto/jQuery) element node.
310      * @returns {View} View object, based on Backbone.View.
311      */
312     V5.View = function (node) {
313         var View = Backbone.View.extend({
314             el: node,
315             bind: function (name, method) {
316                 this[name] = method;
317             },
318             undelegateEvents: function () {

320             }
321         });
322         return new View();

338     var getTemplateNormally = function (name, callback) {
339         var template = V5._templates[name];
340         if (template) {
341             callback(template);
342         } else {
343             $.get("templates/" + name + ".tmpl?_=" + new Date().getTime(), function (templ) {
344                 V5._templates[name] = templ;
345                 callback(templ);
346             });
347         }
348     };
349 
350     var getTemplateOptimized = function (name, callback) {
351         var template = V5._templates[name];
352         if (template) {
353             callback(template);
354         } else {
355             $.get("templates/optimized_combo.tmpl?_=" + new Date().getTime(), function (templ) {
356                 $(templ).find("script").each(function (index, script) {
357                     var templateNode = $(script);
358                     var id = templateNode.attr("id");
359                     V5._templates[id] = templateNode.html();
360                 });
361                 callback(V5._templates[name]);
362             });
363         }
364     };
365 
366     /**
367      * @description Fetch the template file.
368      */
369     V5.getTemplate = function (name, callback) {
370         if (V5.templateMode === "normal") {
371             getTemplateNormally(name, callback);
372         } else {
373             getTemplateOptimized(name, callback);
374         }
375     };

402     var Page = function (module) {
403         // Mixin the eventproxy's prototype

407          * @field {function} initialize
408          */

414         this.shrink = function () {};
415         /**
416          * @description The Reappear method, when View reappear after shrink, this function will be invoked.
417          * @field {function} reappear
418          */
419         this.reappear = function () {};
420         /**
421          * @description The Destroy method, should be invoked manually when necessary.
422          * @field {function} reappear
423          */
424         this.destroy = function () {};
425         /**
426          * @description Parameters, store the parameters, for check the view whether changed.
427          * @field {Array} parameters
428          */
429         this.parameters = null;
430         /**
431          * @description Flag whether enable localization.
432          */
433         this.enableL10N = false;
434         // Merge the module's methods
435         _.extend(this, module);

438     /**
439      * @description Open a view from current column or next column.
440      * @memberOf Page.prototype
441      */
442     Page.prototype.openView = function (hash, blank) {
443         var effectColumn;
444         if (blank) {
445             effectColumn = this.columnIndex + 1;
446         } else {
447             effectColumn = this.columnIndex;
448         }
449         var args = hash.split("/");
450         var viewName = args.shift();
451         V5.trigger("openView", viewName, effectColumn, args);

538     /**
539      * @description Local code.

543     /**
544      * @description All localization resources will be stored at here by locale code.
545      * @namespace Localization resources namespace.
546      */
547     V5.L10N = {};
548 
549     /**
550      * @description Gets localization resources by view name.
551      * @param {string} viewName View name
552      * @param {function} callback Callback that will be invoked when sources is got.
553      */
554     V5.fetchL10N = function (viewName, callback) {
555         var code = V5.langCode;
556         $.getJSON("languages/" + viewName + "_" + code + ".lang?_=" + new Date().getTime(), function (data) {
557             // Sets l10n resources to V5.L10N
558             V5.L10N[code] = V5.L10N[code] || {};
559             _.extend(V5.L10N[code], data);
560             callback(V5.L10N[code]);
561         });
562     };
563 
564     /**
565      * @desciption A wrapper method to localize template with the resources. 
566      * @param {string} template template string.
567      * @param {object} resources resources object.
568      * @returns rendered html string.
569      */
570     V5.localize = function (template, resources) {
571         var settings = {
572                 interpolate : /\{\{(.+?)\}\}/g
573             };
574         var compiled = _.compile(template, settings);
575         return compiled(resources);
576     };
